# 3D可视化技术方案

## 1. 技术概述

### 1.1 目标定位
构建基于Web的工业3D可视化系统，支持标准3D模型导入和实时数据驱动动画，为物联网设备提供直观的三维监控界面。

### 1.2 核心特性
- **3D模型导入**: 支持GLTF/GLB/FBX/OBJ等主流格式
- **数据驱动动画**: 实时传感器数据自动触发3D设备动画
- **工业可视化**: 专为工业设备和工艺流程设计的视觉效果
- **跨平台兼容**: Web端和移动端完整3D功能支持
- **高性能渲染**: 优化的WebGL渲染管线

### 4.2 实时动画引擎
```typescript
class Industrial3DAnimator {
  private scene: THREE.Scene;
  private mixers: Map<string, THREE.AnimationMixer>;
  private deviceMappings: Map<string, DeviceDataMapping>;
  
  constructor(scene: THREE.Scene) {
    this.scene = scene;
    this.mixers = new Map();
    this.deviceMappings = new Map();
  }
  
  // 注册设备映射
  registerDevice(mapping: DeviceDataMapping) {
    this.deviceMappings.set(mapping.deviceId, mapping);
  }
  
  // 处理实时数据更新
  updateDeviceData(deviceId: string, sensorData: any) {
    const mapping = this.deviceMappings.get(deviceId);
    if (!mapping) return;
    
    const device3D = this.scene.getObjectByName(deviceId);
    if (!device3D) return;
    
    // 遍历数据绑定，更新3D属性
    Object.entries(mapping.dataBindings).forEach(([sensorKey, targetPath]) => {
      const value = sensorData[sensorKey];
      if (value !== undefined) {
        this.updateProperty(device3D, targetPath, value, mapping);
      }
    });
  }
  
  // 更新3D对象属性
  private updateProperty(
    device: THREE.Object3D, 
    path: string, 
    value: any, 
    mapping: DeviceDataMapping
  ) {
    const [targetName, propertyPath] = path.split('.');
    const target = device.getObjectByName(targetName);
    
    if (!target) return;
    
    switch (propertyPath) {
      case 'rotation.z':
        // RPM转换为旋转速度
        if (typeof value === 'number') {
          const rotationSpeed = (value / 60) * 2 * Math.PI; // RPM -> rad/s
          target.rotation.z += rotationSpeed * (1/60); // 假设60fps
        }
        break;
        
      case 'material.color':
        // 状态映射到颜色
        const statusColors = {
          'running': 0x00ff00,
          'stopped': 0x888888,
          'alarm': 0xff0000,
          'warning': 0xffff00
        };
        const color = statusColors[value] || 0x888888;
        (target as THREE.Mesh).material.color.setHex(color);
        break;
        
      case 'material.emissive':
        // 温度映射到发光
        if (typeof value === 'number') {
          const intensity = Math.min(value / 100, 1); // 0-100度映射到0-1
          (target as THREE.Mesh).material.emissive.setScalar(intensity * 0.3);
        }
        break;
        
      case 'scale.y':
        // 液位映射到缩放
        if (typeof value === 'number') {
          target.scale.y = value / 100; // 百分比转换
        }
        break;
    }
  }
  
  // 动画循环更新
  update(deltaTime: number) {
    this.mixers.forEach(mixer => mixer.update(deltaTime));
  }
}
```

```typescript
  static createFluidFlow(pipe: THREE.Object3D, flowRate: number) {
    const particles = new THREE.BufferGeometry();

    const material = new THREE.PointsMaterial({
      color: 0x4444ff,
      size: 0.1,
      transparent: true,

      color: 0xffffff,
      size: 0.5,
      transparent: true,
      opacity: 0.3,
      map: this.createSteamTexture()
    });
    
    // 粒子系统实现...
    return new THREE.Points(steamGeometry, steamMaterial);
  }
  
  // 温度热力图
  static applyTemperatureHeatmap(surface: THREE.Mesh, temperature: number) {
    const heatMapShader = {
      uniforms: {
        temperature: { value: temperature },
        minTemp: { value: 0 },
        maxTemp: { value: 100 }
      },
      vertexShader: `
        varying vec2 vUv;
        void main() {
          vUv = uv;
          gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
        }
      `,
      fragmentShader: `
        uniform float temperature;
        uniform float minTemp;
        uniform float maxTemp;
        varying vec2 vUv;
        
        vec3 heatColor(float t) {
          // 蓝色(冷) -> 绿色 -> 黄色 -> 红色(热)
          if (t < 0.33) return mix(vec3(0,0,1), vec3(0,1,0), t * 3.0);
          if (t < 0.66) return mix(vec3(0,1,0), vec3(1,1,0), (t - 0.33) * 3.0);
          return mix(vec3(1,1,0), vec3(1,0,0), (t - 0.66) * 3.0);
        }
        
        void main() {
          float t = (temperature - minTemp) / (maxTemp - minTemp);
          gl_FragColor = vec4(heatColor(t), 1.0);
        }
      `
    };
    
    surface.material = new THREE.ShaderMaterial(heatMapShader);
  }
  
  // 告警闪烁效果
  static createAlarmEffect(device: THREE.Object3D, alarmType: 'warning' | 'critical') {
    const alarmColors = {
      warning: 0xffff00,
      critical: 0xff0000
    };
    
    const originalMaterial = (device as THREE.Mesh).material;
    const alarmMaterial = originalMaterial.clone();
    alarmMaterial.emissive.setHex(alarmColors[alarmType]);
    
    // 闪烁动画
    const blinkTween = new TWEEN.Tween({ intensity: 0 })
      .to({ intensity: 1 }, 500)
      .yoyo(true)
      .repeat(Infinity)
      .onUpdate((obj) => {
        alarmMaterial.emissive.multiplyScalar(obj.intensity);
      });
    
    blinkTween.start();
    return blinkTween;
  }
}
```

### 5.1 渲染优化
```typescript
class PerformanceOptimizer {
  private renderer: THREE.WebGLRenderer;
  private scene: THREE.Scene;
  private camera: THREE.Camera;
  
  constructor(renderer: THREE.WebGLRenderer, scene: THREE.Scene, camera: THREE.Camera) {
    this.renderer = renderer;
    this.scene = scene;
    this.camera = camera;
    this.setupOptimizations();
  }
  
  private setupOptimizations() {
    // 启用自动LOD
    this.enableLOD();
    
    // 视锥体剔除
    this.enableFrustumCulling();
    
    // 遮挡剔除
    this.enableOcclusionCulling();
    
    // 实例化渲染
    this.enableInstancing();
  }
  
  // LOD多级细节
  private enableLOD() {
    const lodDevices = this.scene.children.filter(obj => obj.userData.isDevice);
    
    lodDevices.forEach(device => {
      const lod = new THREE.LOD();
      
      // 高精度模型 (近距离)
      lod.addLevel(device.children[0], 0);
      
      // 中精度模型 (中距离)
      if (device.children[1]) {
        lod.addLevel(device.children[1], 50);
      }
      
      // 低精度模型 (远距离)
      if (device.children[2]) {
        lod.addLevel(device.children[2], 200);
      }
      
      // 替换原对象
      device.parent?.add(lod);
      device.parent?.remove(device);
    });
  }
  
  // 视锥体剔除
  private enableFrustumCulling() {
    const frustum = new THREE.Frustum();
    const cameraMatrix = new THREE.Matrix4();
    
    // 每帧更新视锥体
    this.scene.traverse((object) => {
      if (object.type === 'Mesh') {
        cameraMatrix.multiplyMatrices(this.camera.projectionMatrix, this.camera.matrixWorldInverse);
        frustum.setFromProjectionMatrix(cameraMatrix);
        
        // 检查对象是否在视锥体内
        object.visible = frustum.intersectsObject(object);
      }
    });
  }
  
  // 自适应质量控制
  adaptiveQuality() {
    const fps = this.getCurrentFPS();
    
    if (fps < 20) {
      // 降低质量
      this.renderer.setPixelRatio(Math.min(window.devicePixelRatio * 0.75, 1));
      this.disablePostProcessing();
      this.reduceLightingQuality();
    } else if (fps > 45) {
      // 提升质量
      this.renderer.setPixelRatio(window.devicePixelRatio);
      this.enablePostProcessing();
      this.restoreLightingQuality();
    }
  }
  
  private getCurrentFPS(): number {
    // FPS监控实现
    return performance.now() / 1000;
  }
}
```

```typescript
class MobileOptimizer {
  private isMobile: boolean;
  private deviceCapability: 'low' | 'medium' | 'high';
  
  constructor() {
    this.isMobile = /Android|iPhone|iPad|iPod|BlackBerry|IEMobile|Opera Mini/i.test(navigator.userAgent);
    this.deviceCapability = this.detectDeviceCapability();
  }
  
  // 设备能力检测
  private detectDeviceCapability(): 'low' | 'medium' | 'high' {
    const gl = document.createElement('canvas').getContext('webgl');
    if (!gl) return 'low';
    
    const maxTextures = gl.getParameter(gl.MAX_TEXTURE_IMAGE_UNITS);
    const maxVertexTextures = gl.getParameter(gl.MAX_VERTEX_TEXTURE_IMAGE_UNITS);
    
    if (maxTextures >= 16 && maxVertexTextures >= 4) return 'high';
    if (maxTextures >= 8 && maxVertexTextures >= 2) return 'medium';
    return 'low';
  }
  
  // 移动端配置
  getMobileConfig(): RenderConfig {
    const configs = {
      low: {
        shadowMap: false,
        antialias: false,
        pixelRatio: 0.5,
        maxLights: 2,
        lodBias: 2,
        particleCount: 100
      },
      medium: {
        shadowMap: true,
        antialias: false,
        pixelRatio: 0.75,
        maxLights: 4,
        lodBias: 1,
        particleCount: 500
      },
      high: {
        shadowMap: true,
        antialias: true,
        pixelRatio: 1,
        maxLights: 8,
        lodBias: 0,
        particleCount: 1000
      }
    };
    
    return this.isMobile ? configs[this.deviceCapability] : configs.high;
  }
}
```

```typescript
  switchToView(viewName: string, duration: number = 1000) {
    const view = this.presetViews.get(viewName);

    // 平滑过渡到目标视角
    const startPosition = this.camera.position.clone();
    const startTarget = this.controls.target.clone();
    
    const tween = new TWEEN.Tween({ t: 0 })
      .to({ t: 1 }, duration)
      .easing(TWEEN.Easing.Cubic.InOut)
      .onUpdate((obj) => {
        this.camera.position.lerpVectors(startPosition, view.position, obj.t);
        this.controls.target.lerpVectors(startTarget, view.target, obj.t);
        this.controls.update();
      });
    
    tween.start();
  }
  
  // 自动巡检路径
  startAutoTour(waypoints: CameraView[], interval: number = 5000) {
    let currentIndex = 0;
    
    const tour = () => {
      this.switchToView(waypoints[currentIndex].name);
      currentIndex = (currentIndex + 1) % waypoints.length;
      setTimeout(tour, interval);
    };
    
    tour();
  }
}
```

```typescript
// 设备点击交互
class DeviceInteraction {
  private raycaster: THREE.Raycaster;
  private mouse: THREE.Vector2;
  private scene: THREE.Scene;
  private camera: THREE.Camera;
  
  constructor(scene: THREE.Scene, camera: THREE.Camera, domElement: HTMLElement) {
    this.raycaster = new THREE.Raycaster();
    this.mouse = new THREE.Vector2();
    this.scene = scene;
    this.camera = camera;
    
    domElement.addEventListener('click', this.onMouseClick.bind(this));
    domElement.addEventListener('mousemove', this.onMouseMove.bind(this));
  }
  
  private onMouseClick(event: MouseEvent) {
    // 计算鼠标位置
    const rect = (event.target as HTMLElement).getBoundingClientRect();
    this.mouse.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
    this.mouse.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;
    
    // 射线检测
    this.raycaster.setFromCamera(this.mouse, this.camera);
    const intersects = this.raycaster.intersectObjects(this.scene.children, true);
    
    if (intersects.length > 0) {
      const clickedObject = intersects[0].object;
      const device = this.findDeviceParent(clickedObject);
      
      if (device && device.userData.isDevice) {
        this.showDeviceInfo(device);
      }
    }
  }
  
  private showDeviceInfo(device: THREE.Object3D) {
    const deviceData = device.userData;
    
    // 显示设备信息面板
    const infoPanel = {
      deviceId: deviceData.deviceId,
      deviceType: deviceData.deviceType,
      status: deviceData.currentStatus,
      parameters: deviceData.parameters,
      position: device.position
    };
    
    // 发送事件到React组件
    window.dispatchEvent(new CustomEvent('deviceSelected', {
      detail: infoPanel
    }));
  }
  
  // 设备悬停高亮
  private onMouseMove(event: MouseEvent) {
    // 更新鼠标位置
    const rect = (event.target as HTMLElement).getBoundingClientRect();
    this.mouse.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
    this.mouse.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;
    
    // 射线检测悬停对象
    this.raycaster.setFromCamera(this.mouse, this.camera);
    const intersects = this.raycaster.intersectObjects(this.scene.children, true);
    
    // 清除之前的高亮
    this.clearHighlight();
    
    if (intersects.length > 0) {
      const hoveredObject = intersects[0].object;
      const device = this.findDeviceParent(hoveredObject);
      
      if (device && device.userData.isDevice) {
        this.highlightDevice(device);
        document.body.style.cursor = 'pointer';
      } else {
        document.body.style.cursor = 'default';
      }
    }
  }
  
  private highlightDevice(device: THREE.Object3D) {
    // 添加高亮边框效果
    const outline = new THREE.LineSegments(
      new THREE.EdgesGeometry((device as THREE.Mesh).geometry),
      new THREE.LineBasicMaterial({ color: 0xffff00, linewidth: 2 })
    );
    
    device.add(outline);
    device.userData.outline = outline;
  }
  
  private clearHighlight() {
    this.scene.traverse((object) => {
      if (object.userData.outline) {
        object.remove(object.userData.outline);
        delete object.userData.outline;
      }
    });
  }
}
```

## 7. 数据标签系统

### 7.1 3D数据标签
```typescript
// 3D场景中的浮动数据标签
class Data3DLabel {
  private labelElement: HTMLElement;
  private device3D: THREE.Object3D;
  private camera: THREE.Camera;
  private renderer: THREE.WebGLRenderer;
  
  constructor(device: THREE.Object3D, camera: THREE.Camera, renderer: THREE.WebGLRenderer) {
    this.device3D = device;
    this.camera = camera;
    this.renderer = renderer;
    this.createLabelElement();
  }
  
  private createLabelElement() {
    this.labelElement = document.createElement('div');
    this.labelElement.className = 'data-label-3d';
    this.labelElement.style.cssText = `
      position: absolute;
      background: rgba(0, 0, 0, 0.8);
      color: white;
      padding: 8px 12px;
      border-radius: 4px;
      font-size: 12px;
      pointer-events: none;
      z-index: 1000;
      transition: all 0.3s ease;
    `;
    
    document.body.appendChild(this.labelElement);
  }
  
  // 更新标签位置和内容
  update(deviceData: any) {
    // 计算3D位置对应的屏幕坐标
    const screenPosition = this.get3DScreenPosition();
    
    // 更新标签位置
    this.labelElement.style.left = `${screenPosition.x}px`;
    this.labelElement.style.top = `${screenPosition.y}px`;
    
    // 更新标签内容
    this.labelElement.innerHTML = `
      <div><strong>${deviceData.deviceId}</strong></div>
      <div>状态: ${deviceData.status}</div>
      <div>温度: ${deviceData.temperature}°C</div>
      <div>压力: ${deviceData.pressure} bar</div>
    `;
    
    // 根据距离调整透明度
    const distance = this.camera.position.distanceTo(this.device3D.position);
    const opacity = Math.max(0.1, Math.min(1, 100 / distance));
    this.labelElement.style.opacity = opacity.toString();
  }
  
  private get3DScreenPosition(): { x: number; y: number } {
    const vector = new THREE.Vector3();
    
    // 获取设备世界坐标
    this.device3D.getWorldPosition(vector);
    
    // 转换为屏幕坐标
    vector.project(this.camera);
    
    const canvas = this.renderer.domElement;
    const rect = canvas.getBoundingClientRect();
    
    return {
      x: (vector.x * 0.5 + 0.5) * rect.width + rect.left,
      y: (vector.y * -0.5 + 0.5) * rect.height + rect.top
    };
  }
  
  dispose() {
    if (this.labelElement.parentNode) {
      this.labelElement.parentNode.removeChild(this.labelElement);
    }
  }
}
```
